>  二分插入排序法

```javascript
function binaryInsert(arr) {
    var temp;
    for (var i = 0; i < arr.length; i ++) {
        temp = arr[i];

        var left = 0;
        var right = i - 1;
        while (left <= right) {
            var mid = parseInt((right - left) / 2) + left;
            if (arr[mid] > temp) {
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }

        for (var j = i - 1; j >= left; j --) {
            arr[j + 1] = arr[j]
        }
        arr[left] = temp;
    }
    return arr;
}

console.log(binaryInsert([1,65,53,22,66]))
```
